# Git
git status # Current status on my project
git reset --hard # Revert all my changes
git add -A # Include all changes into scope
git add {path_specific_file/folder} # Include specific changes
git commit -am "<ANY_COMMENT>" # Commit your changes
git pull origin master # Download last version from repository
git push origin master # Send your changes to repository
